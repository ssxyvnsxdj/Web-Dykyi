Програмування

def calculator():
    print("Простий калькулятор на Python")
    a = float(input("Введіть перше число: "))
    b = float(input("Введіть друге число: "))
    операція = input("Виберіть операцію (+, -, *, /): ")
    
    if операція == "+":
        результат = a + b
    elif операція == "-":
        результат = a - b
    elif операція == "*":
        результат = a * b
    elif операція == "/":
        if b != 0:
            результат = a / b
        else:
            return "Ділення на нуль неможливе."
    else:
        return "Невідома операція."
    
    return f"Результат: {результат}"

print(calculator())
